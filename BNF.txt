<programa> ::= 

<funcion> ::= "SubAlgoritmo" <tipo> <identificador> "(" <parametros_funcion> ")" "{"
                <sentencias>  "}" "FinSubAlgoritmo"

<parametros_funcion> ::= <parametro_funcion> | <parametros_funcion> "," <parametro_funcion>

<parametro_funcion> ::= <tipo> <identificador>

<sentencias> ::= <sentencia> ";" | <sentencias> <sentencia>

<sentencia> ::= <asignacion> | <condicional>| <bucle> | <lectura_escritura> | <retorno> | <seleccion>

<asignacion> ::= <identificador> "=" <exp> ";" | <identificador> "=" <identificador> ";"

<condicion> ::= "Si" "(" <exp> ")" "{" <sentencias> "}" "FinSi" |
                "Si" "(" <exp> ")" "{" <sentencias> "}" "Sino" "{" <sentencias> "}" "FinSi"

<bucle> ::= <bucle_hacer_mientras> | <bucle_mientras> | <bucle_para>

<bucle_mientras> ::= "Mientras" "(" <exp> ")" "{" <sentencias> "}"
